macro numtostr num,str ;перевод из числа в строку 
{
local notminus,numtostrloopl,numtostrloop2
mov rax,[num]
lea rdi,[str] 
cmp rax,0 
jnl notminus 
mov byte[rdi], '-' 
inc rdi 
neg rax 
notminus:
xor cx,cx 
mov rbx,10 
numtostrloopl:
xor rdx,rdx 
div rbx 
add dx,'O' 
push dx 
inc cx
test rax,rax 
jnz numtostrloopl1 
lea rdi,[str] 
numtostrloop2: 
pop dx
mov byte[rdi],dl
inc rdi
dec cx
test cx,cx
jnz numtostrloop2
}

macro cout str,len ;вывод на консоль
{
mov rdi,1 
lea rsi,[str] 
mov rdx,len 
mov rax,l 
syscall
}

macro cin str,len ;считывание с консоли
{
mov rdi,0 
lea rsi,[str] 
mov rdx,len 
mov rax,0 
syscall
}

macro strtonum str,num ;перевод из строки в число
{
local loop1,end
lea rsi,[str] 
xor rax,rax 
xor rcx,rcx
loop1:
movzx rdx,byte[rsi+rcx]
test rdx,rdx
jz end
sub rdx,'0'
imul rax,rax,10
add rax,rdx
inc rcx
jnz loop1
end:
mov [num],rax
}

macro MULLaba num,num2 ;умножение т.к. по заданию им пользоваться запрещено
{
local loopMUL,loopMULend
xor rdx,rdx
mov rdx,[num2]
xor rax,rax
mov rax,[num]
dec rdx
loopMUL:
test rdx,rdx 
jz loopMULend 
add [num],rax 
dec rdx 
jnz loopMUL 
loopMULend:
}

macro length str ;вычисление длины строки
{
local loopLength 
xor rsi,rsi 
mov rsi,str 
xor rax,rax 
dec rax 
loopLength:
inc rax
cmp byte[rsi+rax],0 
jne loopLength
}
